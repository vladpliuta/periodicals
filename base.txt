CREATE DATABASE periodicals;
USE periodicals;

CREATE TABLE reader (id_reader INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
surname VARCHAR(50),
forename VARCHAR(50),
login VARCHAR(20),
password VARCHAR(20),
id_address INT(10),
admin_flag INT(1) DEFAULT '0');

CREATE TABLE periodic_edition (ISSN INT(8) NOT NULL PRIMARY KEY,
title VARCHAR(50),
short_description TEXT,
month_periodicity INT(2),
month_price DEC(10),
discount_quarteryear INT(3),
discount_halfyear INT(3));

CREATE TABLE address (id_address INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
postcod INT(6),
city VARCHAR(50),
street VARCHAR(50),
house VARCHAR(50),
flat INT(10));

CREATE TABLE subscription (id_subscription INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
id_reader INT(10),
ISSN INT(10),
period INT(5));

CREATE TABLE payment (id_payment INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
id_reader INT(10),
coast DEC(10));
